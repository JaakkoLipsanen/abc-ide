19  	public String getCode() { return _snippet.getText(); }

21  	/**
22  	 * This function sets the new code AND processes it if needed.
23  	 * For example, when "\n" (aka newline) is inputted, the
24  	 * proper indentation/tabs will be added here
25  	 * @param newValue the value that is to be setted
26  	 * @param caretPosition The position where the cursor/caret was after the new code was inputted. TODO: this sucks,
27  	 *						a better alternative would be that I'd make some kind of "Diff" class which would hold the 
28  	 *						precise differences between the old and new code. Atm this is needed, because it's impossible
29  	 *						to find out where the newline was inputted if there are multiple newlines in a row
30  	 * @return the processed (aka possibly modified) version of the newValue parameter
31  	 */
32  	public String setAndProcessNewCode(String newCode, int caretPosition) { // HMMM!! Should I make "insertCode" function? It'd maybe be clearer!!
33  		if(!newCode.equals(_snippet.getText())) {
34  			String processedCode = _codeFormatter.processNewCode(_snippet.getText(), newCode, caretPosition);
35  			_snippet.setText(processedCode);
36  				
37  			/* todo: syntax highlighting? */
38  			
39  			return processedCode;
40  		}
41  		
42  		return _snippet.getText();
43  	}
