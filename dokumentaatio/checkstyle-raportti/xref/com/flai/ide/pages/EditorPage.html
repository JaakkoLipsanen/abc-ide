1   package com.flai.ide.pages;
2   
3   import com.flai.ide.view.CodeEditorControl;
4   import com.flai.ide.viewmodels.EditorViewModel;
5   import javafx.scene.Node;
6   import javafx.scene.Scene;
7   import javafx.scene.control.Button;
8   import javafx.scene.control.ToolBar;
9   import javafx.scene.layout.BorderPane;
10  
11  /**
12   * EditorPage is the main (and possibly will be the only) page of this
13   * application. It displays the code editor
14   *
15   * @author Jaakko
16   */
17  public class EditorPage implements Page {
18  
19  	private final EditorViewModel _editorViewModel = new EditorViewModel();
20  
21  	@Override
22  	public Scene createScene() {
23  		// creates the main content holder, which will hold all the elements in the page
24  		final BorderPane mainContentHolder = new BorderPane();
25  
26  		mainContentHolder.setTop(createToolBar()); // sets the toolbar to at the top of the window
27  		mainContentHolder.setCenter(createCodeEditor()); // sets the code editor to not be attached to any border (top/left/down/right) of the window
28  
29  		Scene scene = new Scene(mainContentHolder, 600, 800);
30  		scene.getStylesheets().add(EditorPage.class.getResource("/ide-styles.css").toExternalForm());
31  		
32  		return scene;
33  	}
34  
35  	/**
36  	 * Creates the top toolbar (containing new, save, load, run etc buttons)
37  	 *
38  	 * @return the javafx.Node that contains the toolbar
39  	 */
40  	private Node createToolBar() {
41  		final Button testButton = new Button("Run");
42  		testButton.setOnAction(event -> {
43  			/* tadaa */
44  		});
45  
46  		final ToolBar toolBar = new ToolBar(testButton);
47  		return toolBar;
48  	}
49  
50  	/**
51  	 * Creates the text are that contains the code
52  	 *
53  	 * @return the javafx.Node that contains the toolbar
54  	 */
55  	private Node createCodeEditor() {
56  		final CodeEditorControl control = new CodeEditorControl(_editorViewModel);
57  		return control.createNode();
58  	}
59  }
