12  public class CodeSnippetViewModel {
13  
14  	private final CodeSnippet _snippet;
15  	private final CodeFormatter _codeFormatter;
16  	private final CodeSyntaxProcessor _codeSyntaxProcessor;
17  
18  	public CodeSnippetViewModel(CodeSnippet snippet) {
19  		_snippet = snippet;
20  		_codeFormatter = CodeFormatter.create(_snippet.getLanguage());
21  		_codeSyntaxProcessor = CodeSyntaxProcessor.create(_snippet.getLanguage());
22  	}
23  
24  	public String getCode() {
25  		return _snippet.getText();
26  	}
27  
28  	/**
29  	 * This function sets the new code AND processes it if needed. For example,
30  	 * when "\n" (aka newline) is inputted, the proper indentation/tabs will be
31  	 * added here
32  	 *
33  	 * @param newValue the value that is to be setted
34  	 * @param caretPosition The position where the cursor/caret was after the
35  	 * new code was inputted. TODO: this sucks, a better alternative would be
36  	 * that I'd make some kind of "Diff" class which would hold the precise
37  	 * differences between the old and new code. Atm this is needed, because
38  	 * it's impossible to find out where the newline was inputted if there are
39  	 * multiple newlines in a row
40  	 * @return the processed (aka possibly modified) version of the newValue
41  	 * parameter
42  	 */
43  	public String setAndProcessNewCode(String newCode, int caretPosition) { // HMMM!! Should I make "insertCode" function? It'd maybe be clearer!!
44  		if (!newCode.equals(_snippet.getText())) {
45  			String processedCode = _codeFormatter.processNewCode(_snippet.getText(), newCode, caretPosition);
46  			_snippet.setText(processedCode);
47  
48  			/* todo: syntax highlighting? */
49  			return processedCode;
50  		}
51  
52  		return _snippet.getText();
53  	}
54  	
55  	public CodeBlockContainer createSyntaxHighlighting(String code) {
56  		return _codeSyntaxProcessor.processCode(code);
57  	}
