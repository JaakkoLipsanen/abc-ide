17  public interface CodeSyntaxProcessor {

20  	public static CodeSyntaxProcessor create(ProgrammingLanguage language) {
21  		switch (language) {
22  			case JAVA:
23  				return new JavaSyntaxProcessor();
24  		}
25  
26  		throw new IllegalArgumentException("");
27  	}

29  	public enum CodeBlockType {
30  		Keyword,
31  		
32  		SingleLineComment,
33  		MultiLineComment,
34  		
35  		NumberLiteral,
36  		CharLiteral,
37  		StringLiteral,
38  		
39  		OpeningBrace, 
40  		ClosingBrace, 
41  		
42  		Other, // everything else
43  		
44  		// local variable
45  		// parameter variable
46  		// type
47  		// member variable
48  		// method name
49  		// class name
50  		// ^^ these require actual parsing etc and that too hc for this project
51  	}
52  	
53  	public class CodeBlockContainer implements Iterable<CodeBlock> {
54  		private final CodeBlock[] _codeBlocks;
55  		public CodeBlockContainer(Collection<CodeBlock> blocks) {
56  			_codeBlocks = blocks.toArray(new CodeBlock[blocks.size()]);
57  		}
58  		
59  		public int getCount() {
60  			return _codeBlocks.length;
61  		}
62  		
63  		public CodeBlock get(int i) {
64  			assert i >= 0 && i < _codeBlocks.length;
65  			return _codeBlocks[i];
66  		}
67  
68  		@Override
69  		public Iterator<CodeBlock> iterator() {
70  			return Arrays.stream(_codeBlocks).iterator();
71  		}
72  	}
73  	
74  	
75  	public class CodeBlock {
76  		public final int StartIndex;
77  		public final int EndIndex;
78  		
79  		public final CodeBlockType Type;
80  		public final Object Tag; // you can put some metadata here (like what specific keyword it is)
81  		
82  		public CodeBlock(int from, int to, CodeBlockType type) {
83  			this(from, to, type, null);
84  		}
85  		
86  		public CodeBlock(int from, int to, CodeBlockType type, Object tag) {
87  			this.StartIndex = from;
88  			this.EndIndex = to;
89  			
90  			this.Type = type;
91  			this.Tag = tag;
92  		}
93  	}
