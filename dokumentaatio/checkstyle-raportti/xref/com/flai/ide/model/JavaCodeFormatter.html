6   package com.flai.ide.model;
7   
8   import com.flai.ide.StringHelper;
9   import com.flai.ide.StringHelper.TextInsert;
10  import com.flai.ide.model.syntaxhighlight.CodeSyntaxProcessor;
11  import com.flai.ide.model.syntaxhighlight.CodeSyntaxProcessor.CodeBlock;
12  import com.flai.ide.model.syntaxhighlight.CodeSyntaxProcessor.CodeBlockContainer;
13  import com.flai.ide.model.syntaxhighlight.CodeSyntaxProcessor.CodeBlockType;
14  
15  /**
16   *
17   * @author Jaakko
18   */
19  class JavaCodeFormatter implements CodeFormatter {
20  
21  	private final CodeSyntaxProcessor _syntaxProcessor = CodeSyntaxProcessor.create(ProgrammingLanguage.JAVA);
22  	// TODO: when closing brace is inputted, format that also properly
23  	
24  	@Override
25  	public String processNewCode(String oldCode, String newCode, int caretPosition) {
26  		/* If the diff between oldCode and newCode is a newline ("\n"),
27  		 then add a proper indentatation  */
28  
29  		TextInsert insert = StringHelper.isStringDifferenceAnInsertion(oldCode, newCode);
30  		if (insert != null && insert.InsertedText.equals("\n")) { // TODO: it could be "\n\r" or "   \n" or something maybe also?
31  
32  			int indentationLevel = calculateIndentationAtIndex(newCode, insert.StartIndex - 1);
33  			return StringHelper.insert(newCode,
34  					caretPosition,
35  					StringHelper.repeat('\t', indentationLevel));
36  		}
37  
38  		return newCode;
39  	}
40  
41  	/* "{" adds +1 to indentation, "}" -1, as long as they are not inside string or comment */
42  	private int calculateIndentationAtIndex(String code, int index) {
43  		assert index < code.length();
44  
45  		// find amount of "OpeningBrace" code blocks substracted by "ClosingBrace" code blocks at 'index'
46  		CodeBlockContainer codeBlocks = _syntaxProcessor.processCode(code);
47  		
48  		int indentationLevel = 0;
49  		for (CodeBlock block : codeBlocks) {
50  			if(block.StartIndex > index) {
51  				break;
52  			}
53  			
54  			if(block.Type == CodeBlockType.OpeningBrace) {
55  				indentationLevel++;
56  			}
57  			else if(block.Type == CodeBlockType.ClosingBrace) {
58  				indentationLevel--;
59  			}
60  		}
61  		
62  		return indentationLevel >= 0 ? indentationLevel : 0;
63  	}
64  }
