6   package com.flai.ide.model.codecompilers;
7   
8   import java.io.InputStream;
9   import java.io.OutputStream;
10  import java.util.function.Consumer;
11  import javafx.application.Platform;
12  
13  /**
14   * Class that holds information about an program and has ability to run it
15   * @author Jaakko
16   */
17  public final class ProgramInfo {
18  
19  	private final ProgramRunner _runner;
20  
21  	private InputStreamListener _inputListener;
22  	private InputStreamListener _errorListener;
23  	private OutputStreamBroadcaster _outputBroadcaster;
24  	
25  	// returns exit code
26  	private Consumer<Integer> _onFinishedListener;
27  
28  	ProgramInfo(ProgramRunner runner) {
29  		_runner = runner;
30  	}
31  
32  	/**
33  	 * Sets the stdout listener
34  	 * @param listener 
35  	 */
36  	public void setOutputListener(InputStreamListener listener) {
37  		_inputListener = listener;
38  	}
39  
40  	/**
41  	 * Sets the stderr listener 
42  	 * @param listener 
43  	 */
44  	public void setErrorListener(InputStreamListener listener) {
45  		_errorListener = listener;
46  	}
47  
48  	/**
49  	 * sets the stdin 'broadcaster'
50  	 * @param broadcaster 
51  	 */
52  	public void setInputBroadcaster(OutputStreamBroadcaster broadcaster) {
53  		_outputBroadcaster = broadcaster;
54  	}
55  	
56  	/**
57  	 * Sets the callback for when the program has finished running 
58  	 * @param finishedListener 
59  	 */
60  	public void setProgramFinishedListener(Consumer<Integer> finishedListener) {
61  		_onFinishedListener = finishedListener;
62  	}
63  
64  	/**
65  	 * runs/executes the program
66  	 */
67  	public void run() {
68  		new Thread() { // run the app in different thread so that the app doesn't freeze/get blocked
69  
70  			@Override
71  			public void run() {
72  				int exitCode = _runner.run(_inputListener, _errorListener, _outputBroadcaster);
73  				if(_onFinishedListener != null) {
74  					Platform.runLater(() -> {
75  						_onFinishedListener.accept(exitCode);
76  					});
77  				}
78  			}
79  		}.start();
80  	}
81  	
82  	/**
83  	 * stdout/stderr listener
84  	 */
85  	public interface InputStreamListener {
86  		void setStream(InputStream stream);
87  	}
88  
89  	/**
90  	 * stdout 'broadcaster'
91  	 */
92  	public interface OutputStreamBroadcaster {
93  		void setStream(OutputStream stream);
94  	}
95  
96  	/**
97  	 * Common interface for running an program.
98  	 */
99  	public interface ProgramRunner {
100 		// returns exit code
101 		int run(InputStreamListener input, InputStreamListener error, OutputStreamBroadcaster output);
102 	}
103 }
